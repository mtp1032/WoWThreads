98-- @returns running thread, threadId. nil, -1 if thread not found.
99local function getRunningHandle()
100 -- only one thread can be "running"
101 for _, H in ipairs(threadControlBlock) do
102 local state = coroutine.status(H[TH_COROUTINE])
103 if state == "running" then
104 return H, H[TH_UNIQUE_ID]
105 end
106 end
107 -- if we're here, it's because the WoW Client has issued
108 -- this call. That's why there is not A "running" thread
109 -- in the TCB.
110 return nil, nil
111end
112local function createHandle(ticks, threadFunction )
113
114 if ticks < DEFAULT_YIELD_TICKS then
115 ticks = DEFAULT_YIELD_TICKS
116 end
117 THREAD_SEQUENCE_ID = THREAD_SEQUENCE_ID + 1
118
119 local H = {} -- create an empty handle table, H
120
121 H[TH_COROUTINE] = coroutine.create(threadFunction)
122 H[TH_STATUS] = coroutine.status(H[TH_COROUTINE])
123 H[TH_UNIQUE_ID] = THREAD_SEQUENCE_ID
124 H[TH_SIGNAL_STACK] = stackLib.Create()
125 H[TH_DURATION_TICKS] = ticks
126 H[TH_REMAINING_TICKS] = ticks
127 H[TH_ACCUM_YIELD_TICKS] = 0
128 H[TH_LIFETIME_TICKS] = 0
129 H[TH_YIELD_COUNT] = 0
130
131
132 H[TH_CHILDREN] = {}
133 H[TH_PARENT] = nil
134
135 -- Metrics
136
137 local parent_h, parentId = getRunningHandle()
138 if parent_h ~= nil then
139 -- this call is being executed by a WoW Thread. Therefore,
140 -- the running thread is the parent and this handle will be
141 -- the child of the running thread.
142 table.insert(parent_h[TH_CHILDREN], H)
143 H[TH_PARENT] = parent_h
144 end
145 return H
146end

148-- Throws an error if the checks fail
149local function handleIsValid(H)
150 if type(H) ~= "table" then
151 errorMsg = sprintf(L["INVALID_TYPE"], "table", type(H))
152 error(errorMsg)
153 end
154 if type(H[TH_SIGNAL_STACK]) ~= "table" then
155 errorMsg = sprintf("%s", L["HANDLE_ILL_FORMED"])
156 error(errorMsg)
157 end
158 if type(H[TH_COROUTINE]) ~= "thread" then
159 errorMsg = sprintf("%s", L["HANDLE_NOT_A_THREAD"])
160 error(errorMsg)
161 end
162
163 local state = coroutine.status(H[TH_COROUTINE])
164 if state == "dead" then
165 if utils:debuggingIsEnabled() then
166 local threadId = H[TH_UNIQUE_ID]
167 local errorMsg = sprintf(L["INVALID_EXE_CONTEXT"], threadId, state)
168 error(errorMsg)
169 end
170 end
171end

189local function signalIsValid(signal)
190 local isValid = true
191
192 local errorMsg = nil
193 if signal == nil then
194 isValid = false

196 end

210-- RETURNS void
211local function scheduleThreads()
212 ACCUMULATED_TICKS = ACCUMULATED_TICKS + 1
213
214 for i, H in ipairs(threadControlBlock) do
215 assert(H ~= nil, "H was nil")
216
217 -- remove any/all dead threads from the TCB and move
218 -- them into the morgue
219 H[TH_STATUS] = coroutine.status(H[TH_COROUTINE])
220
221 if H[TH_STATUS] == "dead" then
222 H[TH_LIFETIME_TICKS] = ACCUMULATED_TICKS - H[TH_LIFETIME_TICKS]
223 table.remove(threadControlBlock, i)
224 table.insert(morgue, H)
225 end
226
227 if H[TH_STATUS] == "suspended" then
228 -- decrement the remaining tick count and replenish the
229 -- remaining ticks
230 H[TH_REMAINING_TICKS] = H[TH_REMAINING_TICKS] - 1
231 if H[TH_REMAINING_TICKS] == 0 then
232 H[TH_REMAINING_TICKS] = H[TH_DURATION_TICKS]
233 end
234 -- resume the thread
235 if H[TH_LIFETIME_TICKS] == 0 then
236 H[TH_LIFETIME_TICKS] = ACCUMULATED_TICKS
237 end
238 local status, errorMsg = coroutine.resume(H[TH_COROUTINE])
239 if not status then
240 local str = sprintf(L["RESUME_FAILED"], H[TH_UNIQUE_ID])
241 errorMsg = sprintf("%s: %s\n", str, errorMsg)
242 error(errorMsg)
243 end
244 end
245 end
246end

322--- @brief yields the processor to the next thread. Thread context required.
323-- @param None
324-- @returnsNone
325function thread:yield()
326 local beforeYieldTicks = ACCUMULATED_TICKS
327
328 coroutine.yield()
329
330 local H = getRunningHandle()
331 handleIsValid( H )
332 if H[TH_YIELD_COUNT] == nil then
333 local stack = sprintf("%s\n", debugstack(2))
334 error( stack)
335 end
336 H[TH_YIELD_COUNT] = H[TH_YIELD_COUNT] + 1
337 local numYieldTicks = ACCUMULATED_TICKS - beforeYieldTicks
338 H[TH_ACCUM_YIELD_TICKS] = H[TH_ACCUM_YIELD_TICKS] + numYieldTicks
339 beforeTicks = 0
340end

483--- @brief get the metrics required to calculate overhead.
484-- @param (handle) thread_h
485-- @returns entry = { threadId, idealTotalYieldTicks, realTotalYieldTicks }
486function thread:getCongestion(thread_h)
487 local threadId = thread_h[TH_UNIQUE_ID]
488 local suspendedTicks = thread_h[TH_ACCUM_YIELD_TICKS]
489 local lifetimeTicks = thread_h[TH_LIFETIME_TICKS]
490 local yieldCount = thread_h[TH_YIELD_COUNT]
491
492 local runtimeTicks = lifetimeTicks - suspendedTicks
493 local yieldInterval = thread_h[TH_DURATION_TICKS]
494 idealTotalYieldTicks = yieldCount * yieldInterval
495 realTotalYieldTicks = yieldCount * suspendedTicks
496
497 efficiency = idealTotalYieldTicks - (1 - (idealTotalYieldTicks) / realTotalYieldTicks)
